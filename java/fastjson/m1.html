// fastjson
byte[] b1 = JSONObject.toJSONBytes(ipBTaobaoOrder);

// java流
byte[] b1 = ObjectUtil.objectSerialize(ipBTaobaoOrder);
IpBTaobaoOrder b2 = (IpBTaobaoOrder) ObjectUtil.objectDeserialize(b1);

public class ObjectUtil {
    public static Object objectDeserialize(byte[] bytes) throws IOException, ClassNotFoundException {
        ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
        ObjectInputStream ois = new ObjectInputStream(bais);
        ois.close();
        bais.close();
        return ois.readObject();
    }

    public static byte[] objectSerialize(Object object) throws IOException {
        ByteArrayOutputStream baos = new ByteArrayOutputStream();
        ObjectOutputStream oos = new ObjectOutputStream(baos);
        oos.writeObject(object);
        oos.close();
        baos.close();
        return baos.toByteArray();
    }
}
